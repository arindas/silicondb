139  public:

141  const std::unique_ptr<MemberList> &get(MemberListType lt)
142  {
143  // find the list with the given type
144  auto it = std::find_if(begin(),end(),[&lt](const auto &ml) { return ml->listType()==lt; });
145  if (it!=end()) return *it;
146  // or create a new list if it is not found
147  emplace_back(std::make_unique<MemberList>(lt));
148  return back();
149  }
