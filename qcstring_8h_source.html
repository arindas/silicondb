241  QCString stripWhiteSpace() const
242  {
243  int sl = (uint)m_rep.size();
244  if (sl==0 || (!std::isspace(m_rep[0]) && !std::isspace(m_rep[sl-1]))) return *this;
245  int start=0,end=sl-1;
246  while (start<sl && std::isspace(m_rep[start])) start++;
247  if (start==sl) return QCString(); // only whitespace
248  while (end>start && std::isspace(m_rep[end])) end--;
249  return QCString(m_rep.substr(start,1+end-start));
250  }

254  QCString &insert( uint index, const char *s )
255  {
256  uint len = s ? qstrlen(s) : 0;
257  if (len>0)
258  {
259  uint ol = (uint)m_rep.size();
260  if (index>ol) // insert beyond end of string and fill gap with spaces
261  {
262  m_rep.resize(index+len);
263  std::memset(&m_rep[ol],' ',index-ol);
264  std::memcpy(&m_rep[index],s,len+1);
265  }
266  else // insert inside the string
267  {
268  m_rep.insert(index,s);
269  }
270  }
271  return *this;
272  }
