187 template<class T> class CompAccept : public DocNode
188 {
189  public:
190  CompAccept() { m_children.setAutoDelete(TRUE); }
191  void accept(DocVisitor *v)
192  {
193  T *obj = dynamic_cast<T *>(this);
194  v->visitPre(obj);
195  QListIterator<DocNode> cli(m_children);
196  DocNode *n;
197  for (cli.toFirst();(n=cli.current());++cli) n->accept(v);
198  v->visitPost(obj);
199  }
200  const QList<DocNode> &children() const { return m_children; }
201  QList<DocNode> &children() { return m_children; }
202  QString::Direction getTextDir(uint nodeIndex) const
203  {
204  unsigned char resultDir = QString::DirNeutral;
205  for (uint i = nodeIndex; i < m_children.count(); i++)
206  {
207  DocNode* node = m_children.at(i);
208  QString::Direction nodeDir = node->getTextDir();
209  resultDir |= (unsigned char)nodeDir;
210  if (resultDir == QString::DirMixed)
211  return QString::DirMixed;
212  }
213  return static_cast<QString::Direction>(resultDir);
214  }
215  QString::Direction getTextBasicDir(uint nodeIndex) const
216  {
217  for (uint i = nodeIndex; i < m_children.count(); i++)
218  {
219  DocNode* node = m_children.at(i);
220  QString::Direction nodeDir = node->getTextBasicDir();
221  if (nodeDir != QString::DirNeutral)
222  return nodeDir;
223  }
224  return QString::DirNeutral;
225  }
226  virtual QString::Direction getTextDir() const
227  {
228  return getTextDir(0);
229  }
230  virtual QString::Direction getTextBasicDir() const
231  {
232  return getTextBasicDir(0);
233  }
234 
235  protected:
236  QList<DocNode> m_children;
237 };
