203  void addVariable(const QCString &name,ScopedTypeVariant stv)
204  {
205  Scope *scope = m_scopes.empty() ? &m_globalScope : &m_scopes.back();
206  scope->emplace(std::make_pair(name.str(),std::move(stv))); // add it to a list
207  }
208  const ScopedTypeVariant *findVariable(const QCString &name)
209  {
210  const ScopedTypeVariant *result = 0;
211  if (name.isEmpty()) return result;
212 
213  // search from inner to outer scope
214  auto it = std::rbegin(m_scopes);
215  while (it != std::rend(m_scopes))
216  {
217  auto it2 = it->find(name.str());
218  if (it2 != std::end(*it))
219  {
220  result = &it2->second;
221  return result;
222  }
223  ++it;
224  }
225  // nothing found -> also try the global scope
226  auto it2 = m_globalScope.find(name.str());
227  if (it2 != m_globalScope.end())
228  {
229  result = &it2->second;
230  }
231  return result;
232  }
233  bool atGlobalScope() const { return m_scopes.empty(); }
234 
235  private:
236  Scope m_globalScope;
237  std::vector<Scope> m_scopes;
238 };
