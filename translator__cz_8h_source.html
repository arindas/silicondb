697  virtual QCString trWriteList(int numEntries)
698  {
699  QCString result;
700  int i;
701  // the inherits list contain `numEntries' classes
702  for (i=0;i<numEntries;i++)
703  {
704  // use generateMarker to generate placeholders for the class links!
705  result+=generateMarker(i); // generate marker for entry i in the list
706  // (order is left to right)
707 
708  if (i!=numEntries-1) // not the last entry, so we need a separator
709  {
710  if (i<numEntries-2) // not the fore last entry
711  result+=", ";
712  else // the fore last entry
713  result+=" a ";
714  }
715  }
716  return result;
717  }
