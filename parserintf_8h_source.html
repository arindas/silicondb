190  bool registerExtension(const char *extension, const char *parserName)
191  {
192  if (parserName==0 || extension==0) return FALSE;
193 
194  const auto &parserIt = m_parsers.find(parserName);
195  if (parserIt == m_parsers.end()) return FALSE;
196 
197  auto extensionIt = m_extensions.find(extension);
198  if (extensionIt != m_extensions.end()) // extension already exists
199  {
200  m_extensions.erase(extensionIt); // remove it (e.g. user specified extension overrules built in one)
201  }
202  m_extensions.emplace(std::string(extension),parserIt->second); // add new mapping
203  return TRUE;
204  }

242  {
243  QCString ext = QCString(extension).lower().data();
244  if (ext.isEmpty()) ext=".no_extension";
245  auto it = m_extensions.find(ext.data());
246  if (it==m_extensions.end() && ext.length()>4)

248  it = m_extensions.find(ext.left(4).data());
